<!-- Asymptotic Notations -->

1 - Constant
log n - Logarithmic
n - Linear
n^2 - Quadratic
n^3 - Cubic
2^n - Exponential

<!-- Example -->

Algorithm Add(a,b)
{
return a+b; ----------------1
}

f(n) = 1
time complexity = constant

<!-- Example -->

Algorithm Sum(A, n)
{
s=0; -------1
for i =1 to n do ---------n+1
{
s=s+A[i] ----n
}
return s; ------1s
}

f(n) = 2n+3
time complexity = linear

<!-- Example -->

Algorithm MatAdd(A,B,C)
{
for i = 1 to n do --------n+1
{
for j = 1 to n do -------n(n+1)
{
C[i][j] = 0; ---------n^2
for k = 1 to n do ------------n^2(n+1)
{
C[i][j] = A[i][j] + B[i][j] ----------n^3
}
}
}
}


f(n) = 2n^3+3n^2+2n+1
time complexity = cubic
